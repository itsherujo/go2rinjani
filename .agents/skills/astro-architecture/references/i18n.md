## src/i18n/utils.ts (Modernized)

Never use manual `getLangFromUrl` URL extraction. Astro 6 provides `Astro.currentLocale` natively.

```ts
import { defaultLang, type Lang } from "./config";
import { ui, type UIKey } from "./ui";

export function useTranslations(lang: Lang | undefined | string) {
  const safeLang = (lang as Lang) ?? defaultLang;
  return function t(key: UIKey): string {
    return (
      (ui[safeLang] as Record<string, string>)[key] ??
      (ui[defaultLang] as Record<string, string>)[key] ??
      key
    );
  };
}

export function getRouteForLang(path: string, targetLang: Lang): string {
  const clean = path.replace(/^\/(en|zh)/, "") || "/";
  return `/${targetLang}${clean}`;
}
```

## LangSwitcher.tsx (island)

```tsx
import { useStore } from "@nanostores/react";
import { lang } from "../stores/lang";
import { getRouteForLang } from "../i18n/utils";
import type { Lang } from "../i18n/config";

const LANGS: { code: Lang; label: string }[] = [
  { code: "en", label: "EN" },
  { code: "zh", label: "中文" },
];

export default function LangSwitcher({ currentPath }: { currentPath: string }) {
  const currentLang = useStore(lang);

  return (
    <div className="flex gap-2">
      {LANGS.map(({ code, label }) => (
        <a
          key={code}
          href={getRouteForLang(currentPath, code)}
          className={
            currentLang === code ? "font-bold" : "opacity-60 hover:opacity-100"
          }
          aria-label={`Switch to ${label}`}
        >
          {label}
        </a>
      ))}
    </div>
  );
}
```

## Usage in any .astro page (Astro 6 feature)

```astro
---
import { useTranslations } from '../i18n/utils'
import LangSwitcher from '../components/islands/LangSwitcher'

// Instead of manual locals, use Astro 6's builtin `currentLocale`
const currentLang = Astro.currentLocale;
const t = useTranslations(currentLang);
const currentPath = Astro.url.pathname;
---

<h1>{t('hero.cta')}</h1>
<nav>
  <a href={`/${currentLang}/services`}>{t('nav.services')}</a>
  <LangSwitcher currentPath={currentPath} client:load />
</nav>
```
